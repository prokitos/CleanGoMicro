cmd = точка входа
pkg = файлы, библиотеки, код, который используется в других проектах.
api = документация или сваггеры.
config = yaml или env файлы.

internal = внутренняя логика приложения.
  app = здесь собираются логика и зависимости, и запускают приложение.
  models = структуры ответов, таблиц и тому подобное.
  config = инициализация конфигов из yaml файлов.
  services = бизнес логика приложения. (уровень бизнес логики)
  transport = роуты, сервер, rmq	(транспортный уровень)
  database  = методы для взаимодействия с бд (уровень базы данных)


internal 
   транспортный уровень.  пользователь взаимодействует с приложением.  
   уровень бизнес логики. основная логика приложения.
   уровень базы данных.   чтения или запись в базы данных.
